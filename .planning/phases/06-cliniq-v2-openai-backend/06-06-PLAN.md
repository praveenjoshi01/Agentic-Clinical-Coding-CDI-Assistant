  <action>
Create a CLI script at `scripts/build_v2_index.py` that builds the v2 FAISS index using OpenAI text-embedding-3-small embeddings.

The script must:

1. Accept API key via `--api-key` argument OR `OPENAI_API_KEY` environment variable (argument takes priority).
2. Import and configure `OpenAIClient` singleton with the provided key, then validate it.
3. Call `cliniq_v2.rag.build_index.build_faiss_index()` to embed all ~70K ICD-10 descriptions and build the FAISS index.
4. Show progress: batch count, estimated cost (~$0.015 for 70K descriptions at $0.02/M tokens), elapsed time.
5. On success: print index location (`~/.cache/cliniq_v2/icd10_index/`) and total vectors.
6. On failure: catch common errors (invalid API key, rate limit, missing ICD-10 data) and print actionable messages.
7. Add `--check` flag that only checks whether the index already exists (prints status, exits 0 if exists, 1 if not).

Script structure:
```python
#!/usr/bin/env python3
"""Build FAISS index for ClinIQ v2 using OpenAI text-embedding-3-small."""

import sys
import os
import time
import argparse
from pathlib import Path

PROJECT_ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(PROJECT_ROOT))

from cliniq_v2.config import INDEX_DIR

def check_index():
    """Check if v2 FAISS index exists."""
    index_path = INDEX_DIR / "icd10.faiss"
    metadata_path = INDEX_DIR / "icd10_metadata.json"
    if index_path.exists() and metadata_path.exists():
        import faiss
        index = faiss.read_index(str(index_path))
        print(f"v2 FAISS index exists: {index.ntotal} vectors at {index_path}")
        return True
    print(f"v2 FAISS index NOT found at {INDEX_DIR}")
    print("Run: python scripts/build_v2_index.py --api-key YOUR_KEY")
    return False

def build_index(api_key: str):
    """Build the v2 FAISS index."""
    from cliniq_v2.api_client import OpenAIClient
    from cliniq_v2.rag.build_index import build_faiss_index

    client = OpenAIClient()
    client.configure(api_key)
    if not client.validate_key():
        print("ERROR: Invalid OpenAI API key.")
        sys.exit(1)

    print("Building ClinIQ v2 FAISS index (text-embedding-3-small, 1536d)...")
    print("Estimated cost: ~$0.015 for ~70K ICD-10 descriptions")
    print(f"Output directory: {INDEX_DIR}")

    start = time.perf_counter()
    index, codes = build_faiss_index()
    elapsed = time.perf_counter() - start

    print(f"\nIndex built successfully in {elapsed:.1f}s")
    print(f"  Vectors: {index.ntotal}")
    print(f"  Dimension: {index.d}")
    print(f"  Location: {INDEX_DIR}")

def main():
    parser = argparse.ArgumentParser(description="Build FAISS index for ClinIQ v2")
    parser.add_argument("--api-key", help="OpenAI API key (or set OPENAI_API_KEY env var)")
    parser.add_argument("--check", action="store_true", help="Check if index exists (don't build)")
    args = parser.parse_args()

    if args.check:
        sys.exit(0 if check_index() else 1)

    api_key = args.api_key or os.environ.get("OPENAI_API_KEY")
    if not api_key:
        print("ERROR: Provide --api-key or set OPENAI_API_KEY environment variable")
        sys.exit(1)

    build_index(api_key)

if __name__ == "__main__":
    main()
```

Follow this structure. Use `print()` for output (not logging) for user-friendly CLI experience. Handle KeyboardInterrupt gracefully.
